            try:
                with candidate.open("r", encoding="utf-8") as f:
                    data = yaml.safe_load(f)

                return data

def store_credential(username: str, secret: str, source: str, cred_type: str = "password") -> None:
    entry = {"username": username, "secret": secret, "source": source, "type": cred_type}
    if entry not in CRED_STORE:
        CRED_STORE.append(entry)
        print(f"[CRED] Stored: {username} / {secret} ({cred_type}) from {source}")

        print(f"  #{i}  [{c['type']}]  {c['username']} : {c['secret']}")

def have_bin(name: str) -> bool:
    """Check system PATH and common pip local install paths."""
    if shutil.which(name) is not None:
        return True
    for extra in ["/usr/local/bin", "/root/.local/bin", os.path.expanduser("~/.local/bin")]:
        if Path(extra, name).exists():
            return True
    return False

def yes_no(prompt: str, default: bool = False) -> bool:
    suffix = "[Y/n]" if default else "[y/N]"
    ans = input(f"{prompt} {suffix}: ").strip().lower()
    if not ans:
        return default
    return ans == "y"


def prompt_until_valid(prompt: str, validator) -> str:
    while True:
        value = input(prompt).strip()
        try:
            return validator(value)
        except ValueError as e:
            print(f"[!] {e}")


def validate_ip(value: str) -> str:
    try:
        ipaddress.ip_address(value)
        return value
    except ValueError:
        raise ValueError("Expected single IP in format X.X.X.X")


def validate_network(value: str) -> str:
    try:
        net = ipaddress.ip_network(value, strict=False)
        return str(net)
    except ValueError:
        raise ValueError("Expected subnet in format X.X.X.0/24")

def run_streaming_command(
    cmd: List[str],
    label: str = "",
    cwd: Optional[Path] = None,
    combined_log: Optional[Path] = None,
    timeout_seconds: Optional[int] = None,
) -> int:
    if label:
        print(f"\n=== {label} ===")
    print("[+] Running:", " ".join(cmd))
    append_log(combined_log, f"\n=== {label or 'Command'} ===\n{' '.join(cmd)}\n\n")

    with subprocess.Popen(
        cmd,
        stdout=subprocess.PIPE,
        stderr=subprocess.STDOUT,
        text=True,
        cwd=str(cwd) if cwd else None,
        bufsize=1,
        universal_newlines=True,
    ) as proc:
        timed_out = False

        def _kill_after(secs: int) -> None:
            nonlocal timed_out
            import time
            time.sleep(secs)
            if proc.poll() is None:
                timed_out = True
                print(f"\n[!] Timeout ({secs}s) for: {label or cmd[0]}. Terminating.")
                try:
                    proc.terminate()
                except Exception:
                    pass

        if timeout_seconds:
            t = threading.Thread(target=_kill_after, args=(timeout_seconds,), daemon=True)
            t.start()

        assert proc.stdout is not None
        for line in proc.stdout:
            print(line, end="")
            append_log(combined_log, line)

        rc = proc.wait()
        return -1 if timed_out else rc


def run_or_resume_oA(
    cmd: List[str],
    base_path: Path,
    label: str,
    cwd: Optional[Path] = None,
    combined_log: Optional[Path] = None,
) -> int:
    xml   = Path(str(base_path) + ".xml")
    nmap  = Path(str(base_path) + ".nmap")
    gnmap = Path(str(base_path) + ".gnmap")

    def non_empty(p: Path) -> bool:
        return p.exists() and p.is_file() and p.stat().st_size > 0

    def xml_ok(p: Path) -> bool:
        if not non_empty(p):
            return False
        try:
            ET.parse(str(p))
            return True
        except ET.ParseError:
            return False

    if non_empty(nmap) and non_empty(gnmap) and xml_ok(xml):
        print(f"[+] Resume: valid Nmap output found for {base_path.name}, skipping.")
        append_log(combined_log, f"[+] Resume skip (valid): {base_path}\n")
        return 0

    if xml.exists() or nmap.exists() or gnmap.exists():
        print(f"[!] Resume: partial/corrupt outputs for {base_path.name}; re-running.")
        append_log(combined_log, f"[!] Resume rerun (partial/corrupt): {base_path}\n")

    return run_streaming_command(cmd, label=label, cwd=cwd, combined_log=combined_log)

def parse_nmap_xml(xml_file: Path) -> Dict:
    result = {"hosts": []}
    if not xml_file.exists() or xml_file.stat().st_size == 0:
        return result
    try:
        tree = ET.parse(xml_file)
    except ET.ParseError:
        return result
    root = tree.getroot()

    for host in root.findall("host"):
        state_el = host.find("status")
        state = state_el.get("state", "unknown") if state_el is not None else "unknown"

        address = "unknown"
        addrs = host.findall("address")
        for addr in addrs:
            if addr.get("addrtype") == "ipv4":
                address = addr.get("addr", "unknown")
                break
        if address == "unknown" and addrs:
            address = addrs[0].get("addr", "unknown")

        hostnames = [x.get("name") for x in host.findall("./hostnames/hostname") if x.get("name")]

        ports = []
        for port in host.findall("./ports/port"):
            st = port.find("state")
            if st is None or st.get("state") != "open":
                continue
            svc = port.find("service")
            scripts = []
            for s in port.findall("script"):
                sid = s.get("id", "")
                sout = s.get("output", "")
                scripts.append({"id": sid, "output": sout})
                # Auto-extract common anonymous access signals
                if sid == "ftp-anon" and "Anonymous FTP login allowed" in sout:
                    store_credential("anonymous", "anonymous", f"FTP {address}:{port.get('portid','')}", "anonymous")
                if sid == "redis-info" and "# Server" in sout:
                    store_credential("", "(no auth)", f"Redis {address}:{port.get('portid','')}", "no-auth")

            # Extract http-title for WordPress title_contains matching
            http_title = ""
            for s in scripts:
                if s["id"] == "http-title":
                    http_title = s["output"]
                    break

            ports.append({
                "port":       port.get("portid", ""),
                "protocol":   port.get("protocol", ""),
                "service":    svc.get("name", "")      if svc is not None else "",
                "product":    svc.get("product", "")   if svc is not None else "",
                "version":    svc.get("version", "")   if svc is not None else "",
                "extrainfo":  svc.get("extrainfo", "") if svc is not None else "",
                "http_title": http_title,
                "scripts":    scripts,
            })

        result["hosts"].append({
            "address":    address,
            "state":      state,
            "hostnames":  hostnames,
            "open_ports": ports,
        })

    return result


def build_html(xml_file: Path, html_file: Path, log_file: Optional[Path] = None) -> None:
    if not have_bin("xsltproc"):
        print("[!] xsltproc not found; skipping HTML generation.")

    run_streaming_command(cmd, label=f"Generate HTML: {html_file.name}", combined_log=log_file)


def merge_nmap_xmls(xml_paths: Iterable[Path], output_xml: Path) -> bool:
    import copy
    xml_list = [p for p in xml_paths if p.exists() and p.stat().st_size > 0]
    if not xml_list:
        return False

    base_tree = None
    base_root = None
    base_path = None
    for p in xml_list:
        try:
            base_tree = ET.parse(p)
            base_root = base_tree.getroot()
            base_path = p
            break
        except ET.ParseError:
            continue

    if base_tree is None or base_root is None:
        return False

    def host_key(host_el: ET.Element) -> str:
        addrs = host_el.findall("address")
        for addr in addrs:
            if addr.get("addrtype") == "ipv4":
                return addr.get("addr", "unknown")
        if addrs:
            return addrs[0].get("addr", "unknown")
        return "unknown"

    host_index = {host_key(h): h for h in base_root.findall("host")}

    for p in xml_list:
        if base_path and p.resolve() == base_path.resolve():
            continue
        try:
            root = ET.parse(p).getroot()
        except ET.ParseError:
            continue

        for host in root.findall("host"):
            key = host_key(host)
            if key not in host_index:
                base_root.append(copy.deepcopy(host))
                host_index[key] = base_root.findall("host")[-1]
                continue

            target_host = host_index[key]
            target_ports = target_host.find("ports")
            if target_ports is None:
                target_ports = ET.SubElement(target_host, "ports")

            existing_ports = {(pt.get("protocol"), pt.get("portid")) for pt in target_ports.findall("port")}
            source_ports = host.find("ports")
            if source_ports is None:
                continue

            for port in source_ports.findall("port"):
                p_key = (port.get("protocol"), port.get("portid"))
                if p_key in existing_ports:
                    continue
                target_ports.append(copy.deepcopy(port))
                existing_ports.add(p_key)

    base_tree.write(output_xml, encoding="utf-8", xml_declaration=True)
    return True


def service_buckets(parsed: Dict) -> Dict[str, List[Tuple[str, str]]]:
    buckets: Dict[str, List[Tuple[str, str]]] = {}
    for host in parsed.get("hosts", []):
        ip = host["address"]
        for p in host["open_ports"]:
            svc = p["service"] or "unknown"
            buckets.setdefault(svc, []).append((ip, p["port"]))
    return buckets


def generate_mermaid(parsed: Dict) -> str:
    lines = ["flowchart TD", " A[Start] --> B[Initial Nmap Scan]"]
    idx = 0
    for host in parsed.get("hosts", []):
        host_node = f"H{idx}"
        lines.append(f' B --> {host_node}["{host["address"]}"]')
        j = 0
        for p in host["open_ports"]:
            port_node = f"P{idx}_{j}"
            label = f'{p["port"]}/{p["protocol"]} {p["service"]}'.replace('"', "'")
            lines.append(f' {host_node} --> {port_node}["{label}"]')
            svc = (p["service"] or "").lower()
            if "http" in svc:
                lines.append(f' {port_node} --> W{idx}_{j}["Web enum"]')
            elif svc in {"microsoft-ds", "netbios-ssn"}:
                lines.append(f' {port_node} --> S{idx}_{j}["SMB enum"]')
            elif svc == "ftp":
                lines.append(f' {port_node} --> F{idx}_{j}["FTP review"]')
            elif svc == "snmp":
                lines.append(f' {port_node} --> N{idx}_{j}["SNMP walk"]')
            elif svc == "ldap":
                lines.append(f' {port_node} --> L{idx}_{j}["LDAP review"]')
            elif svc in {"nfs", "mountd"}:
                lines.append(f' {port_node} --> NF{idx}_{j}["NFS enum"]')
            elif svc == "smtp":
                lines.append(f' {port_node} --> SM{idx}_{j}["SMTP enum"]')
            j += 1
        idx += 1
    return "```mermaid\n" + "\n".join(lines) + "\n```"

FREQ_SCORE = {
    "very_high":   5,
    "high":        4,
    "medium_high": 3,
    "medium":      2,
    "low_medium":  1,
    "low":         1,
}


def score_parsed_against_playbook(parsed: Dict, playbook: Optional[Dict]) -> List[Dict]:
    if not playbook:
        return []

    fingerprint_map   = playbook.get("service_fingerprint_map", [])
    foothold_patterns = {fp["id"]: fp for fp in playbook.get("foothold_patterns", [])}
    scoring           = playbook.get("simple_scoring_model", {})
    signal_bonus      = scoring.get("signal_bonus", {})
    results = []

    for host in parsed.get("hosts", []):
        ip = host["address"]
        for p in host["open_ports"]:
            svc        = (p["service"]    or "").lower()
            product    = (p["product"]    or "").lower()
            version    = (p["version"]    or "").lower()
            port       = p["port"]
            http_title = (p.get("http_title") or "").lower()

            for fm in fingerprint_map:
                match  = fm.get("match", {})
                fm_svc = (match.get("service") or "").lower()
                if fm_svc and fm_svc not in svc:
                    continue
                prod_contains = match.get("product_contains", [])
                if prod_contains and not any(pc.lower() in product for pc in prod_contains):
                    continue
                port_in = match.get("port_in", [])
                if port_in and int(port) not in [int(x) for x in port_in]:
                    continue
                # FIX: title_contains was previously ignored
                title_contains = match.get("title_contains", "")
                if title_contains and title_contains.lower() not in http_title:
                    continue

                base_priority = fm.get("priority", 1)
                likely_paths  = fm.get("likely_paths", [])

                for path_id in likely_paths:
                    fp         = foothold_patterns.get(path_id, {})
                    freq       = fp.get("estimated_exam_frequency", "low")
                    base_score = FREQ_SCORE.get(freq, 1)
                    bonus = 0
                    if version:
                        bonus += signal_bonus.get("version_exposed", 0)
                    if product:
                        bonus += signal_bonus.get("service_default_banner", 0)
                    score = base_priority + base_score + bonus
                    results.append({
                        "ip":           ip,
                        "port":         port,
                        "service":      p["service"],
                        "product":      p["product"],
                        "version":      p["version"],
                        "pattern_id":   path_id,
                        "pattern_name": fp.get("name", path_id),
                        "score":        score,
                        "likely_next":  fp.get("likely_next_steps", []),
                        "enum_cmds":    fp.get("enumeration_commands", fm.get("recommended_enumeration", [])),
                    })

    seen = {}
    for r in results:
        key = (r["ip"], r["port"], r["pattern_id"])
        if key not in seen or r["score"] > seen[key]["score"]:
            seen[key] = r
    return sorted(seen.values(), key=lambda x: x["score"], reverse=True)


def print_triage_list(ranked: List[Dict], output_md: Optional[Path] = None) -> None:
    if not ranked:
        print("[+] No playbook triage data (playbook not loaded or no matches).")
        return

    lines = []
    lines.append("\n" + "=" * 60)
    lines.append("  TRIAGE PRIORITY LIST  (score = priority + freq + signals)")
    lines.append("=" * 60)

    top  = [r for r in ranked if r["score"] >= 6]
    rest = [r for r in ranked if r["score"] < 6]

    def fmt_entry(r: Dict, rank: int) -> str:
        out  = f"\n#{rank}  [{r['score']:>2}]  {r['ip']}:{r['port']}  ({r['service']})\n"
        out += f"      Pattern : {r['pattern_id']} — {r['pattern_name']}\n"
        if r["product"] or r["version"]:
            out += f"      Banner  : {r['product']} {r['version']}\n".rstrip() + "\n"
        if r["enum_cmds"]:
            out += f"      Commands: {r['enum_cmds'][0]}\n"
        if r["likely_next"]:
            out += f"      Next    : {r['likely_next'][0]}\n"
        return out

    if top:
        lines.append("\n  HIGH CONFIDENCE (score >= 6) — pursue these first:\n")
        for i, r in enumerate(top, 1):
            lines.append(fmt_entry(r, i))
    if rest:
        lines.append("\n  LOWER PRIORITY (score < 6) — revisit if top paths fail:\n")
        for i, r in enumerate(rest, len(top) + 1):
            lines.append(fmt_entry(r, i))

    lines.append("=" * 60)
    output = "\n".join(lines)
    print(output)

    if output_md and output_md.exists():
        with output_md.open("a", encoding="utf-8") as f:
            f.write("\n\n## Triage Priority List\n\n```\n")
            f.write(output)
            f.write("\n```\n")

def generate_markdown_summary(
    title: str,
    target: str,
    scan_type: str,
    stamp: str,
    xml_files: List[Path],
    output_md: Path,
    notes: Optional[List[str]] = None,
) -> None:
    lines = [
        f"# {title}", "",
        f"- **Scan Type:** {scan_type}",
        f"- **Target:** `{target}`",
        f"- **Timestamp:** `{stamp}`", "",
    ]
    if notes:
        lines += ["## Notes", ""]
        for note in notes:
            lines.append(f"- {note}")
        lines.append("")

    lines += ["## Results", ""]
    merged_parsed = {"hosts": []}
    for xml in xml_files:
        parsed = parse_nmap_xml(xml)
        merged_parsed["hosts"].extend(parsed["hosts"])
        lines.append(f"### {xml.name}")
        lines.append("")
        if not parsed["hosts"]:
            lines += ["- No parseable hosts found.", ""]
            continue
        for host in parsed["hosts"]:
            lines.append(f"#### Host `{host['address']}`")
            if host["hostnames"]:
                lines.append(f"- Hostnames: {', '.join(host['hostnames'])}")
            if not host["open_ports"]:
                lines += ["- No open ports parsed.", ""]
                continue
            lines.append("- Open ports:")
            for p in host["open_ports"]:
                detail = " ".join(x for x in [p["service"], p["product"], p["version"], p["extrainfo"]] if x).strip()
                lines.append(f"  - `{p['port']}/{p['protocol']}` {detail}".rstrip())
                for s in p["scripts"][:2]:
                    if s["id"] or s["output"]:
                        lines.append(f"    - script `{s['id']}`: {s['output'][:150].replace(chr(10), ' ')}")
            lines.append("")

    lines += ["## Attack Path Summary", "", generate_mermaid(merged_parsed), ""]
    lines += ["## Service Overview", ""]
    buckets = service_buckets(merged_parsed)
    if buckets:
        for svc, items in sorted(buckets.items()):
            item_text = ", ".join(f"{ip}:{port}" for ip, port in items[:10])
            lines.append(f"- **{svc}**: {item_text}")
    else:
        lines.append("- No services parsed.")
    output_md.write_text("\n".join(lines), encoding="utf-8")

NSE_DEFAULT  = "default"

NSE_SMB      = ("smb-security-mode,smb2-security-mode,smb-os-discovery,"
                "smb-vuln-ms17-010,smb-enum-shares,smb-vuln-cve2009-3103")

# FIX: added mincvss filter to reduce vulners noise
NSE_VULNERS  = "vulners --script-args mincvss=6.0"


def nse_for_ports(open_ports: List[Dict]) -> str:
    scripts = {NSE_DEFAULT}
    for p in open_ports:
        svc = (p["service"] or "").lower()
        if "http" in svc or p["port"] in {"80", "443", "8080", "8443"}:
            scripts.add(NSE_HTTP)
        if svc in {"microsoft-ds", "netbios-ssn"} or p["port"] == "445":
            scripts.add(NSE_SMB)
        if svc == "ssh" or p["port"] == "22":
            scripts.add(NSE_SSH)
        if Path("/usr/share/nmap/scripts/vulners.nse").exists():
            scripts.add(NSE_VULNERS)
    return ",".join(scripts)

FOLLOW_UP_TIMEOUT = 300


def follow_up_http(ip: str, port: str, svc: str, out_dir: Path, log_file: Optional[Path]) -> None:
    svc_l       = (svc or "").lower()
    https_ports = {"443", "8443", "9443"}
    is_https    = ("https" in svc_l) or (port in https_ports)
    scheme      = "https" if is_https else "http"
    url         = f"{scheme}://{ip}" if port in {"80", "443"} else f"{scheme}://{ip}:{port}"

    if have_bin("whatweb"):
        run_streaming_command(["whatweb", "-a", "3", url],
                              label=f"WEB whatweb {ip}:{port}",
                              combined_log=log_file, timeout_seconds=FOLLOW_UP_TIMEOUT)
    if have_bin("nikto"):
        run_streaming_command(["nikto", "-h", url],
                              label=f"WEB nikto {ip}:{port}",
                              combined_log=log_file, timeout_seconds=FOLLOW_UP_TIMEOUT)

    # FIX: cascading wordlist search instead of silent skip
    wordlist = best_wordlist(WORDLIST_PRIORITY)
    if have_bin("ffuf") and wordlist:
        run_streaming_command(
            ["ffuf", "-u", f"{url}/FUZZ", "-w", wordlist,
             "-mc", "200,201,204,301,302,307,401,403,405",
             "-t", "50", "-c"],
            label=f"WEB ffuf dir {ip}:{port}", combined_log=log_file,
            timeout_seconds=FOLLOW_UP_TIMEOUT)
        # Extension sweep
        run_streaming_command(
            ["ffuf", "-u", f"{url}/FUZZ", "-w", wordlist,
             "-e", ".php,.asp,.aspx,.txt,.bak,.zip,.sql,.conf,.log",
             "-mc", "200,201,204,301,302,307,401,403",
             "-t", "50", "-c"],
            label=f"WEB ffuf ext {ip}:{port}", combined_log=log_file,
            timeout_seconds=FOLLOW_UP_TIMEOUT)
    elif have_bin("gobuster") and wordlist:
        run_streaming_command(
            ["gobuster", "dir", "-u", url, "-w", wordlist,
             "-x", "php,asp,aspx,txt,bak,zip,sql", "--no-error"],
            label=f"WEB gobuster {ip}:{port}", combined_log=log_file,
            timeout_seconds=FOLLOW_UP_TIMEOUT)

    if have_bin("wpscan"):
        run_streaming_command(
            ["wpscan", "--url", url, "--enumerate", "vp,vt,u", "--no-banner"],
            label=f"WEB wpscan {ip}:{port}", combined_log=log_file,
            timeout_seconds=FOLLOW_UP_TIMEOUT)

def service_follow_up(parsed: Dict, workspace: Path, log_file: Optional[Path]) -> None:
    follow_dir = workspace / "follow_up"
    follow_dir.mkdir(exist_ok=True)
    seen = set()
    for host in parsed.get("hosts", []):
        ip = host["address"]
        for p in host["open_ports"]:
            svc  = (p["service"] or "").lower()
            port = p["port"]
            key  = (ip, port, svc)
            if key in seen:
                continue
            seen.add(key)

            is_http = ("http" in svc) or (
                port in {"80", "443", "8080", "8443", "9443"} and ("http" in svc or svc == ""))

            if is_http:
                follow_up_http(ip, port, svc, follow_dir, log_file)
            elif svc in {"microsoft-ds", "netbios-ssn"} or port == "445":
                follow_up_smb(ip, follow_dir, log_file)
            elif svc == "snmp" or port in {"161", "162"}:
                follow_up_snmp(ip, follow_dir, log_file)
            elif svc == "ftp" or port == "21":
                follow_up_ftp(ip, port, follow_dir, log_file)
            elif svc == "ldap" or port in {"389", "636", "3268", "3269"}:
                follow_up_ldap(ip, log_file)
            elif svc == "redis" or port == "6379":
                follow_up_redis(ip, log_file)
            # NEW dispatches — FIX for all previously missing services
            elif svc in {"nfs", "mountd"} or port == "2049":
                follow_up_nfs(ip, follow_dir, log_file)
            elif svc == "smtp" or port in {"25", "465", "587"}:
                follow_up_smtp(ip, port, log_file)
            elif svc == "domain" or port == "53":
                follow_up_dns(ip, log_file)
            elif svc in {"ms-sql-s", "ms-sql", "mssql"} or port == "1433":
                follow_up_mssql(ip, port, log_file)
            elif svc == "mysql" or port == "3306":
                follow_up_mysql(ip, port, log_file)
            elif svc in {"wsman", "winrm"} or port in {"5985", "5986"}:
                follow_up_winrm(ip, port, log_file)

def exploit_confirm(ranked: List[Dict], workspace: Path, log_file: Optional[Path]) -> None:
    """
    Non-destructive exploit confirmation for high-confidence triage findings.
    Runs safe PoC checks (NSE scripts, banner grabs, showmount, redis PING)
    to confirm exploitability without causing impact.
    """
    if not ranked:
        print("[+] No ranked findings to confirm.")
        return

    high_conf = [r for r in ranked if r["score"] >= 6]
    if not high_conf:
        print("[+] No high-confidence findings (score >= 6) to confirm.")
        return

    print(f"\n[+] Running exploit confirmation for {len(high_conf)} high-confidence finding(s)...")
    confirm_log = workspace / "exploit_confirm.log"

    confirmed: List[str] = []
    for r in high_conf:
        pid = r["pattern_id"]
        fn  = PATTERN_TO_CONFIRM.get(pid)
        if fn:
            fn(r, confirm_log)
            confirmed.append(f"{r['ip']}:{r['port']} [{pid}]")
        else:
            print(f"[+] No automated confirmation for pattern: {pid} — manual check required.")

    # Always check SMB signing for any SMB finding
    smb_ips = {r["ip"] for r in ranked if "smb" in r.get("service", "").lower()
               or r["port"] == "445"}
    for ip in smb_ips:
        _confirm_smb_signing(ip, confirm_log)

    print("\n[+] Exploit confirmation complete.")
    if confirmed:
        print("    Confirmed checks run:")
        for c in confirmed:
            print(f"      {c}")
    print(f"    Log: {confirm_log}")

# Credential spray (NEW)
# ─────────────────────────────────────────────

def credential_spray(parsed: Dict, workspace: Path, log_file: Optional[Path]) -> None:
    """
    Spray accumulated credentials across discovered services.
    Non-destructive: uses netexec/crackmapexec with --no-bruteforce flag.
    """
    if not CRED_STORE:
        print("[!] No credentials in session store. Skipping spray.")
        return

    cme = "netexec" if have_bin("netexec") else ("crackmapexec" if have_bin("crackmapexec") else None)
    if not cme:
        print("[!] netexec/crackmapexec not found. Cannot spray.")
        return

    spray_log = workspace / "cred_spray.log"
    real_creds = [c for c in CRED_STORE if c.get("type") not in ("anonymous", "no-auth")]
    if not real_creds:
        print("[!] Only anonymous/no-auth credentials found. Nothing to spray.")
        return

    services_seen: Dict[str, List[str]] = {}  # protocol -> [ip, ...]
    for host in parsed.get("hosts", []):
        ip = host["address"]
        for p in host["open_ports"]:
            svc  = (p["service"] or "").lower()
            port = p["port"]
            if svc in {"microsoft-ds", "netbios-ssn"} or port == "445":
                services_seen.setdefault("smb", []).append(ip)
            if svc in {"wsman", "winrm"} or port in {"5985", "5986"}:
                services_seen.setdefault("winrm", []).append(ip)
            if svc == "ssh" or port == "22":
                services_seen.setdefault("ssh", []).append(ip)

    print(f"\n[+] Spraying {len(real_creds)} credential(s) across {sum(len(v) for v in services_seen.values())} service target(s).")
    print("[!] NOTE: using --no-bruteforce — testing exact username:password pairs only.")

    for proto, ips in services_seen.items():
        for cred in real_creds:
            for ip in ips:
                cmd = [cme, proto, ip,
                       "-u", cred["username"], "-p", cred["secret"],
                       "--no-bruteforce"]
                run_streaming_command(cmd, label=f"SPRAY {proto} {ip} {cred['username']}",
                                      combined_log=spray_log, timeout_seconds=30)

def two_phase_scan(
    target: str,
    workspace: Path,
    stamp: str,
    log_file: Optional[Path],
    label_prefix: str = "IP",
    extra_flags: Optional[List[str]] = None,
) -> List[Path]:
    extra = extra_flags or []
    xmls: List[Path] = []

    base1 = workspace / f"{label_prefix}_{sanitize_target(target)}_{stamp}_phase1"
    cmd1 = [
        "nmap", "-sS", "-sV", "-T4", "--open",
        "--min-rate", "5000",
        "--script", f"{NSE_DEFAULT},{NSE_HTTP},{NSE_SMB},{NSE_SSH}",
        "--stats-every", "10s",
        "-oA", str(base1),
    ] + extra + [target]
    run_or_resume_oA(cmd1, base1, f"{label_prefix} Phase 1 — Top-1000 TCP + NSE",
                     cwd=workspace, combined_log=log_file)
    xml1 = Path(str(base1) + ".xml")
    if xml1.exists():
        xmls.append(xml1)

    base2      = workspace / f"{label_prefix}_{sanitize_target(target)}_{stamp}_phase2"
    parsed1    = parse_nmap_xml(xml1) if xml1.exists() else {"hosts": []}
    all_ports1 = [p for h in parsed1["hosts"] for p in h["open_ports"]]
    nse_full   = nse_for_ports(all_ports1)

    cmd2 = [
        "nmap", "-sS", "-sV", "-Pn", "-T4", "-p-",
        "--min-rate", "5000",
        "--script", nse_full,
        "--stats-every", "10s",
        "-oA", str(base2),
    ] + extra + [target]
    print("\n[+] Phase 2 (full -p-) starting. Phase 1 results are already available above.")
    run_or_resume_oA(cmd2, base2, f"{label_prefix} Phase 2 — Full TCP -p-",
                     cwd=workspace, combined_log=log_file)
    xml2 = Path(str(base2) + ".xml")
    if xml2.exists():
        xmls.append(xml2)

    if yes_no("Run UDP sweep against critical ports?", default=True):
        udp_xml = run_udp_sweep(target, workspace, stamp, log_file)
        if udp_xml:
            xmls.append(udp_xml)

    return xmls


# ─────────────────────────────────────────────
# Post-processing
# ─────────────────────────────────────────────

def post_process(
    xmls: List[Path],
    workspace: Path,
    title: str,
    target: str,
    stamp: str,
    scan_type: str,
    log_file: Optional[Path] = None,
    extra_notes: Optional[List[str]] = None,
) -> None:
    combined_xml  = workspace / f"{scan_type}_{sanitize_target(target)}_{stamp}_combined.xml"
    combined_html = workspace / f"{scan_type}_{sanitize_target(target)}_{stamp}_combined.html"
    combined_md   = workspace / f"{scan_type}_{sanitize_target(target)}_{stamp}_combined.md"

    if len(xmls) == 1:
        combined_xml = xmls[0]
    elif len(xmls) > 1:
        merge_nmap_xmls(xmls, combined_xml)

    build_html(combined_xml, combined_html, log_file=log_file)

    notes = [
        f"Workspace: `{workspace}`",
        f"HTML: `{combined_html.name if combined_html.exists() else 'not generated'}`",
    ] + (extra_notes or [])

    generate_markdown_summary(
        title=title, target=target, scan_type=scan_type,
        stamp=stamp, xml_files=[combined_xml], output_md=combined_md, notes=notes)

    parsed = parse_nmap_xml(combined_xml)
    generate_obsidian_host_notes(parsed, workspace / "obsidian_notes")

    ranked = score_parsed_against_playbook(parsed, PLAYBOOK)
    print_triage_list(ranked, output_md=combined_md)

    # Exploit confirmation
    if yes_no("\nRun exploit confirmation checks (non-destructive PoC)?", default=True):
        exploit_confirm(ranked, workspace, log_file)

    # Service follow-up
    if yes_no("Run automatic service-based follow-up enumeration?", default=False):
        service_follow_up(parsed, workspace, log_file)

    # Credential spray if any creds were accumulated
    if CRED_STORE:
        print_cred_store()
        if yes_no("Spray accumulated credentials across discovered services?", default=False):
            credential_spray(parsed, workspace, log_file)

    maybe_run_obs(combined_xml, workspace)
    print(f"\n[+] Workspace: {workspace}")

def discover_live_hosts(network: str, workspace: Path, stamp: str, log_file: Optional[Path]) -> List[str]:
    base = workspace / f"Discovery_{sanitize_target(network)}_{stamp}"

def detailed_subnet_scan() -> None:
    target    = prompt_until_valid("Enter subnet (X.X.X.0/24): ", validate_network)
    stamp     = now_stamp()
    workspace = make_workspace("Full", target, stamp)
    log_file  = workspace / "run.log"

    if not yes_no("Use parallel host-based mode?", default=True):
        xmls = two_phase_scan(target, workspace, stamp, log_file, label_prefix="Full")
        post_process(xmls=xmls, workspace=workspace, title="Detailed Subnet Scan Summary",
                     target=target, stamp=stamp, scan_type="Full", log_file=log_file)
        return

    live_hosts = discover_live_hosts(target, workspace, stamp, log_file)
    if not live_hosts:
        print("[!] No live hosts found.")
        return

    worker_raw = input("Parallel worker count [default 4]: ").strip()
    try:
        workers = max(1, int(worker_raw)) if worker_raw else 4
    except ValueError:
        workers = 4

    all_xmls: List[Path] = []
    total = len(live_hosts)
    print(f"[+] Parallel scan of {total} hosts with {workers} workers.")

    with ThreadPoolExecutor(max_workers=workers) as ex:
        futures   = {ex.submit(host_full_scan, ip, workspace, stamp): ip for ip in live_hosts}
        completed = 0
        for future in as_completed(futures):
            ip = futures[future]
            try:
                host, xml_paths = future.result()
                completed += 1
                print(f"[+] Completed {completed}/{total}: {host}")
                all_xmls.extend(xml_paths)
            except Exception as e:
                completed += 1
                print(f"[!] Error on {ip}: {e}")

    post_process(
        xmls=all_xmls, workspace=workspace,
        title="Detailed Subnet Parallel Scan Summary",
        target=target, stamp=stamp, scan_type="Full", log_file=log_file,
        extra_notes=[f"Live hosts: `{len(live_hosts)}`", f"Workers: `{workers}`"])

    steps = [
        ("Step 1/5 - AD Services", [
            "nmap", "-sS", "-sV", "-Pn", "-T4",
            "-p", "53,88,135,389,445,464,636,3268,3269,5985,5986",

            "--script", "ldap* and not brute",
            "--stats-every", "10s",
            "-oA", str(workspace / f"{base}_ldap"), target]),
        ("Step 3/5 - SMB Info + Signing", [
            "nmap", "-p", "445",
            "--script", "smb-security-mode,smb2-security-mode,smb-os-discovery",
            "--stats-every", "10s",
            "-oA", str(workspace / f"{base}_smb"), target]),
        ("Step 4/5 - SMB Enum", [
            "nmap", "-p", "445",
            "--script", "smb-enum-shares,smb-enum-users,smb-enum-groups",
            "--stats-every", "10s",
            "-oA", str(workspace / f"{base}_smb_enum"), target]),
        ("Step 5/5 - Kerberos", [
            "nmap", "-p", "88",
            "--script", "krb5-enum-users",
            "--stats-every", "10s",
            "-oA", str(workspace / f"{base}_kerb"), target]),
    ]

    xmls: List[Path] = []
    for label, cmd in steps:
        out_base = Path(cmd[cmd.index("-oA") + 1])
        run_or_resume_oA(cmd, out_base, label, cwd=workspace, combined_log=log_file)
        xml = Path(str(out_base) + ".xml")
        if xml.exists():
            xmls.append(xml)

        udp_xml = run_udp_sweep(target, workspace, stamp, log_file)
        if udp_xml:
            xmls.append(udp_xml)

        run_streaming_command([cme, "smb", target, "--shares", "-u", "", "-p", ""],
                              label="AD null-session share listing",
                              combined_log=log_file, timeout_seconds=60)
        run_streaming_command([cme, "smb", target, "--users", "-u", "", "-p", ""],
                              label="AD null-session user listing",
                              combined_log=log_file, timeout_seconds=60)

    post_process(
        xmls=xmls, workspace=workspace, title="AD Scan Summary",
        target=target, stamp=stamp, scan_type="AD", log_file=log_file,
        extra_notes=[
            f"Combined log: `{log_file.name}`",
            "Steps: services, LDAP, SMB info+signing, SMB enum, Kerberos"])

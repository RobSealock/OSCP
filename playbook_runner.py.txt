#!/usr/bin/env python3
"""Playbook runner (planner-first).

Loads the OSCP playbook YAML and produces:
  - prioritized hypotheses (foothold_patterns) based on observed services/signals
  - recommended enumeration modules by stack/service
  - a structured checklist output (markdown or json)

This runner intentionally defaults to *planning* and *note output*.
If you later add an execution mode, keep it separate and explicit.
"""

from __future__ import annotations

import argparse
import json
import re
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple

try:
    import yaml  # PyYAML
except Exception as e:
    raise SystemExit("Missing dependency: pyyaml. Install with: pip install pyyaml") from e


# -------------------------
# Data structures
# -------------------------

@dataclass
class ServiceObservation:
    service: str
    port: Optional[int] = None
    product: Optional[str] = None
    title: Optional[str] = None
    banner: Optional[str] = None
    tags: List[str] = field(default_factory=list)


@dataclass
class MatchRule:
    raw: Dict[str, Any]

    def priority(self) -> int:
        return int(self.raw.get("priority", 0))

    def likely_paths(self) -> List[str]:
        return list(self.raw.get("likely_paths", []))

    def recommended_enumeration(self) -> List[str]:
        return list(self.raw.get("recommended_enumeration", []))

    def matches(self, obs: ServiceObservation) -> bool:
        m = self.raw.get("match", {})
        if m.get("service") and m.get("service") != obs.service:
            return False

        # Optional port_in
        ports = m.get("port_in")
        if ports is not None and obs.port is not None:
            if obs.port not in ports:
                return False

        # Optional product/title contains
        prod_contains = m.get("product_contains")
        if prod_contains is not None:
            if obs.product is None:
                return False
            if isinstance(prod_contains, list):
                if not any(p.lower() in obs.product.lower() for p in prod_contains):
                    return False
            else:
                if str(prod_contains).lower() not in obs.product.lower():
                    return False

        title_contains = m.get("title_contains")
        if title_contains is not None:
            if obs.title is None:
                return False
            if str(title_contains).lower() not in obs.title.lower():
                return False

        return True


# -------------------------
# Loading & validation
# -------------------------

REQUIRED_TOP_KEYS = [
    "schema_version",
    "service_fingerprint_map",
    "web_stack_enum_modules",
    "foothold_patterns",
]


def load_playbook(path: Path) -> Dict[str, Any]:
    data = yaml.safe_load(path.read_text(encoding="utf-8"))
    if not isinstance(data, dict):
        raise ValueError("Playbook root must be a mapping/dict")

    missing = [k for k in REQUIRED_TOP_KEYS if k not in data]
    if missing:
        raise ValueError(f"Playbook missing required keys: {', '.join(missing)}")

    # Normalize a few optional sections
    data.setdefault("reference_catalog", {})
    data.setdefault("reporting_helpers", {})
    return data


# -------------------------
# Matching logic
# -------------------------


def build_rules(playbook: Dict[str, Any]) -> List[MatchRule]:
    rules = []
    for r in playbook.get("service_fingerprint_map", []):
        if isinstance(r, dict) and "match" in r:
            rules.append(MatchRule(r))
    # Highest priority first
    rules.sort(key=lambda x: x.priority(), reverse=True)
    return rules


def score_paths(playbook: Dict[str, Any], matched_paths: List[str], observed_signals: List[str]) -> List[Tuple[str, int, Dict[str, Any]]]:
    """Return list of (pattern_id, score, pattern) sorted by score desc."""
    model = playbook.get("simple_scoring_model", {})
    base_scores = model.get("base_scores", {})
    bonus = model.get("signal_bonus", {})

    patterns = {p.get("id"): p for p in playbook.get("foothold_patterns", []) if isinstance(p, dict)}

    results: List[Tuple[str, int, Dict[str, Any]]] = []
    for pid in matched_paths:
        p = patterns.get(pid)
        if not p:
            # path referenced by rules but not defined
            continue
        freq = p.get("estimated_exam_frequency", "low")
        base = int(base_scores.get(freq, 0))
        score = base

        # Simple signal bonus: if your observed_signals include keywords that match any of the pattern signals,
        # add a conservative 'admin_panel_discovered' etc only if you tag them explicitly.
        # (Runner leaves signal inference to user inputs.)
        for s in observed_signals:
            # user can provide structured tags like: credentials_recovered, upload_form_present
            if s in bonus:
                score += int(bonus[s])

        results.append((pid, score, p))

    results.sort(key=lambda t: t[1], reverse=True)
    return results


def recommend_modules(playbook: Dict[str, Any], obs: ServiceObservation) -> List[Dict[str, Any]]:
    """Return enum modules that apply to a given observation."""
    out: List[Dict[str, Any]] = []
    for stack in playbook.get("web_stack_enum_modules", []):
        if not isinstance(stack, dict):
            continue
        detect = stack.get("detect_if", {})

        # generic http/https match
        any_service = detect.get("any_service")
        if any_service and obs.service in any_service:
            out.extend(stack.get("modules", []))
            continue

        # title/banner/whatweb identifiers
        title_contains = detect.get("title_contains")
        if title_contains and obs.title and str(title_contains).lower() in obs.title.lower():
            out.extend(stack.get("modules", []))
            continue

        banner_contains = detect.get("banner_contains")
        if banner_contains and obs.banner and str(banner_contains).lower() in obs.banner.lower():
            out.extend(stack.get("modules", []))
            continue

        whatweb_identifies = detect.get("whatweb_identifies")
        if whatweb_identifies and obs.product and str(whatweb_identifies).lower() in obs.product.lower():
            out.extend(stack.get("modules", []))
            continue

    return out


# -------------------------
# Output helpers
# -------------------------


def render_markdown(target: str, observations: List[ServiceObservation], ranked: List[Tuple[str, int, Dict[str, Any]]], module_map: Dict[str, List[Dict[str, Any]]]) -> str:
    lines = []
    lines.append(f"# Playbook Plan for {target}\n")
    lines.append("## Observed Services\n")
    for o in observations:
        port = f":{o.port}" if o.port else ""
        lines.append(f"- **{o.service}{port}**" + (f" — product: {o.product}" if o.product else ""))
    lines.append("\n## Top Hypotheses (Ranked)\n")
    if not ranked:
        lines.append("No matching foothold patterns found for the provided observations.")
    for pid, score, p in ranked[:10]:
        lines.append(f"### {pid} — score {score}\n")
        lines.append(f"**{p.get('name','')}**\n")
        if p.get("likely_next_steps"):
            lines.append("Next steps (checklist):")
            for s in p["likely_next_steps"]:
                lines.append(f"- {s}")
        if p.get("enumeration_commands"):
            lines.append("Suggested enumeration commands (from playbook):")
            for c in p["enumeration_commands"]:
                lines.append(f"- `{c}`")
        lines.append("")

    lines.append("\n## Enumeration Modules by Service\n")
    for key, modules in module_map.items():
        lines.append(f"### {key}\n")
        for m in modules:
            lines.append(f"- **{m.get('name','module')}**")
            for c in m.get("commands", []):
                lines.append(f"  - `{c}`")
        lines.append("")

    return "\n".join(lines) + "\n"


# -------------------------
# CLI
# -------------------------


def parse_observations(path: Optional[Path], inline: Optional[List[str]]) -> List[ServiceObservation]:
    obs: List[ServiceObservation] = []

    def parse_line(line: str) -> ServiceObservation:
        # Format examples:
        #   http:80 product=nginx title="Welcome" tags=admin_panel_discovered
        #   microsoft-ds:445 tags=anonymous_access
        parts = line.strip().split()
        if not parts:
            raise ValueError("empty observation")
        svc_part = parts[0]
        if ":" in svc_part:
            service, port_s = svc_part.split(":", 1)
            port = int(port_s)
        else:
            service, port = svc_part, None
        kw = {"product": None, "title": None, "banner": None, "tags": []}
        for p in parts[1:]:
            if "=" not in p:
                continue
            k, v = p.split("=", 1)
            v = v.strip().strip('"').strip("'")
            if k == "tags":
                kw["tags"] = [t for t in re.split(r"[,;]", v) if t]
            elif k in ("product", "title", "banner"):
                kw[k] = v
        return ServiceObservation(service=service, port=port, product=kw["product"], title=kw["title"], banner=kw["banner"], tags=kw["tags"]) 

    if path:
        for line in path.read_text(encoding="utf-8").splitlines():
            line = line.strip()
            if not line or line.startswith("#"):
                continue
            obs.append(parse_line(line))

    if inline:
        for line in inline:
            obs.append(parse_line(line))

    return obs


def main() -> None:
    ap = argparse.ArgumentParser(description="Playbook runner (planner-first)")
    ap.add_argument("--playbook", required=True, help="Path to playbook YAML")
    ap.add_argument("--target", required=True, help="Target label (IP/hostname)")
    ap.add_argument("--obs-file", help="Path to observations file")
    ap.add_argument("--obs", action="append", help="Inline observation line; can be repeated")
    ap.add_argument("--out", default="plan.md", help="Output file (md or json)")

    args = ap.parse_args()

    playbook = load_playbook(Path(args.playbook))
    rules = build_rules(playbook)

    observations = parse_observations(Path(args.obs_file) if args.obs_file else None, args.obs)

    matched_paths: List[str] = []
    observed_signals: List[str] = []
    module_map: Dict[str, List[Dict[str, Any]]] = {}

    for o in observations:
        observed_signals.extend(o.tags)
        for r in rules:
            if r.matches(o):
                matched_paths.extend(r.likely_paths())
        # modules
        key = f"{o.service}{':' + str(o.port) if o.port else ''}"
        module_map[key] = recommend_modules(playbook, o)

    # Deduplicate while keeping order
    seen = set()
    matched_paths = [p for p in matched_paths if not (p in seen or seen.add(p))]

    ranked = score_paths(playbook, matched_paths, observed_signals)

    out_path = Path(args.out)
    if out_path.suffix.lower() == ".json":
        payload = {
            "target": args.target,
            "observations": [o.__dict__ for o in observations],
            "ranked_hypotheses": [
                {"id": pid, "score": score, "name": pat.get("name"), "next_steps": pat.get("likely_next_steps", [])}
                for pid, score, pat in ranked
            ],
            "modules": module_map,
        }
        out_path.write_text(json.dumps(payload, indent=2), encoding="utf-8")
    else:
        out_path.write_text(render_markdown(args.target, observations, ranked, module_map), encoding="utf-8")


if __name__ == "__main__":
    main()

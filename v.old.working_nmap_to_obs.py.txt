#!/usr/bin/env python3
"""nmap_to_obs.py

Convert Nmap XML output (-oX) into the simple observation lines consumed by playbook_runner.py.

Why XML?
- It's structured and designed for programmatic parsing.

This script:
- Reads one Nmap XML file
- Emits one obs line per open port in the form:
    service:port product=<product/version string if available>

It stays *mapping-only*: no scanning, no execution of tools.
"""

from __future__ import annotations

import argparse
import xml.etree.ElementTree as ET
from pathlib import Path


def safe_get(el, path, attr=None, default=None):
    found = el.find(path)
    if found is None:
        return default
    if attr is None:
        return found.text if found.text is not None else default
    return found.get(attr, default)


def normalize_service_name(name: str) -> str:
    # Keep nmap-style service names where possible.
    # Map common aliases to your playbook match keys.
    mapping = {
        'https': 'http',          # treat https as http for matching modules
        'ssl/http': 'http',
        'msrpc': 'rpc',
        'netbios-ssn': 'netbios-ssn',
        'microsoft-ds': 'microsoft-ds',
        'ms-wbt-server': 'ms-wbt-server',
    }
    return mapping.get(name, name)


def build_product(service_el: ET.Element) -> str | None:
    # Nmap <service> attributes include product, version, extrainfo, name.
    product = service_el.get('product')
    version = service_el.get('version')
    extrainfo = service_el.get('extrainfo')

    parts = [p for p in [product, version, extrainfo] if p]
    if parts:
        return ' '.join(parts)
    return None


def main() -> None:
    ap = argparse.ArgumentParser(description='Convert Nmap XML (-oX) to playbook observations')
    ap.add_argument('--xml', required=True, help='Path to nmap XML file')
    ap.add_argument('--out', default='obs.txt', help='Output obs file')
    args = ap.parse_args()

    xml_path = Path(args.xml)
    out_path = Path(args.out)

    tree = ET.parse(xml_path)
    root = tree.getroot()

    lines = []

    # Iterate hosts -> ports -> port entries
    for host in root.findall('host'):
        status = safe_get(host, 'status', 'state', default='unknown')
        if status != 'up':
            continue

        for port in host.findall('./ports/port'):
            proto = port.get('protocol', '')
            portid = port.get('portid')
            state = safe_get(port, 'state', 'state', default='unknown')
            if state != 'open':
                continue

            service_el = port.find('service')
            if service_el is None:
                svc_name = 'unknown'
                product = None
            else:
                svc_name = normalize_service_name(service_el.get('name', 'unknown'))
                product = build_product(service_el)

            # Observation format: service:port [product=...]
            line = f"{svc_name}:{portid}"
            if product:
                # Quote product if it contains spaces
                line += f" product=\"{product}\""
            lines.append(line)

    out_path.write_text('\n'.join(lines) + ('\n' if lines else ''), encoding='utf-8')


if __name__ == '__main__':
    main()
